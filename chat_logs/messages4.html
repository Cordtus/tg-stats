import base64
import json
import sys
import struct

def decode_result(encoded_message):
    data = base64.b64decode(encoded_message)
    offset = 0

    def read_uint32_be(data, offset):
        value = struct.unpack(">I", data[offset:offset+4])[0]
        return value, offset + 4

    def read_uint64_be(data, offset):
        value = struct.unpack(">Q", data[offset:offset+8])[0]
        return value, offset + 8

    def read_string(data, offset, length):
        value = data[offset:offset+length].decode()
        return value, offset + length

    result, offset = read_string(data, offset, len(data) - offset)
    return result

const axios = require('axios');
const fs = require('fs').promises;

// Path to the gist file containing the API response data
const GIST_PATH = './gistfile1.txt';

// Function to fetch pool data from the gist file
async function fetchPoolData() {
  try {
    const data = await fs.readFile(GIST_PATH, 'utf-8');
    const parsedData = JSON.parse(data);

    if (parsedData.result && parsedData.result.data && parsedData.result.data.json && parsedData.result.data.json.items) {
      return parsedData.result.data.json.items;
    } else {
      console.error('Unexpected response structure:', parsedData);
      return [];
    }
  } catch (error) {
    console.error('Error fetching pool data:', error);
    return [];
  }
}

// Function to safely parse RatePretty JSON strings
function parseRatePretty(ratePrettyString) {
  try {
    const parsed = JSON.parse(ratePrettyString);
    return parseFloat(parsed.rate);
  } catch (e) {
    return 0;
  }
}

// Function to calculate APR breakdown
function calculateAprBreakdown(pool) {
  if (!pool.aprBreakdown) {
    return {
      total: 0,
      swapFees: 0,
      superfluid: 0,
      osmosis: 0,
      boost: 0,
    };
  }

  const swapFees = parseRatePretty(pool.aprBreakdown.swapFee) || 0;
  const superfluidRewards = parseRatePretty(pool.aprBreakdown.superfluid) || 0;
  const osmosisIncentives = parseRatePretty(pool.aprBreakdown.osmosis) || 0;
  const externalBoost = parseRatePretty(pool.aprBreakdown.boost) || 0;

  const totalApr = swapFees + superfluidRewards + osmosisIncentives + externalBoost;

  return {
    total: totalApr,
    swapFees: swapFees,
    superfluid: superfluidRewards,
    osmosis: osmosisIncentives,
    boost: externalBoost,
  };
}

// Main function to verify APR calculations
async function verifyAprCalculations() {
  const pools = await fetchPoolData();

  pools.forEach(pool => {
    const aprBreakdown = calculateAprBreakdown(pool);
    console.log(`Pool ID: ${pool.id}`);
    console.log('APR Breakdown:', aprBreakdown);
    console.log('--------------------------------');
  });
}

// Run the verification script
verifyAprCalculations();
